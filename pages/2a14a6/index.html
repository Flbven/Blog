字符串分组

# 字符串分组 (opens new window)

# 方法一 并查集

本题目words中的字符串标注：只包含小写英文字母、每个字母至多出现一次，因此可以想到使用一个int类型变量来表示每一个字符串。

同时可以对题目中的三种操作进行简化：

  • 将变量中的某一位1变成0
  • 将变量中的某一位0变成1
  • 将变量中的某一位1变为0，同时某一位0变为1

使用并查集进行分类字符串。

  • 时间复杂度 α
  • 空间复杂度 
struct UnionFind {
  int n;
  vector<int> parent, size;
  UnionFind(int _n): n(_n), parent(n), size(n, 1) {
    for(int i = 0 ; i < n ; ++ i) parent[i] = i;
  }
  int find(int x) {
    if(parent[x] == x)
      return x;
    return parent[x] = find(parent[x]);
  }
  void connect(int x, int y) {
    int fx = find(x), fy = find(y);
    if(fx != fy) {
      if(size[fx] > size[fy]) {
        parent[fy] = fx;
        size[fx] += size[fy];
      } else {
        parent[fx] = fy;
        size[fy] += size[fx];
      }
    }
  }
};
class Solution {
public:
    vector<int> groupStrings(vector<string>& words) {
      int n = words.size();
      unordered_map<int, int> mp;
      UnionFind uf(n);
      for(int i = 0 ; i < n ; ++ i) {
        // 二进制表示该字符串
        int bit = 0;
        for(char c : words[i])
          bit ^= 1 << (c - 'a');

        if(mp.count(bit)) {
          uf.connect(mp[bit], i);
        } else {
          // 操作1
          for(int j = 0 ; j < 26 ; ++ j) {
            int nbit = bit ^ (1 << j);
            if(mp.count(nbit))
              uf.connect(mp[nbit], i);
          }
          // 操作2
          for(int j = 0 ; j < 26 ; ++ j) {
            if((bit & (1 << j)) == 0) continue;
            for(int k = 0 ; k < 26 ; ++ k) {
              if(bit & (1 << k)) continue;
              int nbit = bit ^ (1 << j) ^ (1 << k);
              if(mp.count(nbit))
                uf.connect(mp[nbit], i);
            }
          }
          mp[bit] = i;
        }

      }
      int groups = 0, largest = 0;
      for(int i = 0 ; i < n ; ++ i) {
        if(uf.find(i) == i) {
          groups ++ ;
          largest = max(largest, uf.size[i]);
        }
      }

      return {groups, largest};
    }
};
